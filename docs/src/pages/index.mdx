# Amplify Spark (React)

## Default Authenticator

```jsx
import * as Spark from "@aws-amplify/spark-react";
const { SparkContext } = Spark

<SparkContext.Provider value={Spark}>
  <Authenticator>{({ user }) => (user ? <App /> : <SignIn />)}</Authenticator>
</SparkContext.Provider>;
```

## Custom Components

```jsx
<SparkContext.Provider
  value={{
    ...Spark,
    classNames: {
      "Authenticator.Wrapper": "custom-heading",
    },
    components: {
      Heading: () => {
        return <h1>Howdy!</h1>;
      },
    },
  }}
>
  <Authenticator>{({ user }) => (user ? <App /> : <SignIn />)}</Authenticator>
</SparkContext.Provider>
```
